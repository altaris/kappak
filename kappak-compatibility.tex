	\newcommand{\squarediagram}[8]{
		\[ \begin{tikzpicture}
			\matrix (m) [matrix of math nodes,
					row sep = \@tRowsep em,
					column sep = \@tColsep em,
					text height = 1.5 ex,
					text depth = 0.25 ex,
					ampersand replacement = \&] {
				#1 \& #2 \\
				#3 \& #4 \\
			};
			\path [>=stealth, \@tDiagSqUA, \@tDiagSqUL] (m-1-1) edge node [above] {$#5$} (m-1-2);
			\path [>=stealth, \@tDiagSqLA, \@tDiagSqLL] (m-1-1) edge node [left]  {$#6$} (m-2-1);
			\path [>=stealth, \@tDiagSqRA, \@tDiagSqRL] (m-1-2) edge node [right] {$#7$} (m-2-2);
			\path [>=stealth, \@tDiagSqDA, \@tDiagSqDL] (m-2-1) edge node [above] {$#8$} (m-2-2);
		\end{tikzpicture}
		\]
		\def\@tDiagSqUA{->}\def\@tDiagSqLA{->}\def\@tDiagSqRA{->}\def\@tDiagSqDA{->}\def\@tDiagSqUL{solid}\def\@tDiagSqLL{solid}\def\@tDiagSqRL{solid}\def\@tDiagSqDL{solid}
