\section{Math wubba lubba dub dub}

\begin{lemma}[Yoneda] \label{lemma:yoneda}
For $\catCC$ a locally small category, and $\pSh (\catCC) = [ \catCC^\op, \Set ]$ we have an isomorphism natural in $F \in \pSh (\catCC)$ and $C \in \catCC$:
\[ \pSh (\catCC) (\yoneda_C, F) \cong FC . \]
\end{lemma}

\begin{corollary} \label{corollary:yoneda}
We have an embedding:
\[ \yoneda : \catCC \longincl \pSh (\catCC) \]
\end{corollary}

We see that \cref{lemma:yoneda} is very useful, and so is
\cref{corollary:yoneda}.

\section{Diagrams: a friendly tour}

Here's simple square diagram:

\begin{verbatim}
\squarediagram{A}{B}{C}{D}{\alpha}{\beta}{\gamma}{\delta}
\end{verbatim}
\[ \squarediagram{A}{B}{C}{D}{\alpha}{\beta}{\gamma}{\delta} \]

Let's apply styles:
\begin{verbatim}
\diagramarrows{->>}{<-}{-}{}
\diagramlines{}{--}{}{=}
\squarediagram{A}{B}{C}{D}{\alpha}{\beta}{\gamma}{\delta}
\end{verbatim}
\diagramarrows{->>}{<-}{-}{}
\diagramlines{}{--}{}{=}
\[ \squarediagram{A}{B}{C}{D}{\alpha}{\beta}{\gamma}{\delta} \]

Don't forget to remove styles! (once you don't need them anymore)
\begin{verbatim}
\rds
\end{verbatim}
\rds

Here are the other diagrams:
\begin{verbatim}
\pushoutdiagram{A}{B}{C}{D}{\alpha}{\beta}{\gamma}{\delta}
\end{verbatim}
\[ \pushoutdiagram{A}{B}{C}{D}{\alpha}{\beta}{\gamma}{\delta} \]

Note the default direction of the arrows. You can change that by applying some style:
\begin{verbatim}
\diagramarrows{<-}{}{}{}
\triangleUdiagram{A}{B}{C}{\alpha}{\beta}{\gamma}
\end{verbatim}
\diagramarrows{<-}{}{}{}
\[ \triangleUdiagram{A}{B}{C}{\alpha}{\beta}{\gamma} \]

Heck, let's apply more styles:
\begin{verbatim}
\diagramlines{..}{}{=}{}
\triangleUdiagram{A}{B}{C}{\alpha}{\beta}{\gamma}
\end{verbatim}
\diagramlines{..}{}{=}{}
\[ \triangleUdiagram{A}{B}{C}{\alpha}{\beta}{\gamma} \]

Notice how the first arrow is still reversed. That's why you want to use
\begin{verbatim}
\rds
\end{verbatim}
\rds
